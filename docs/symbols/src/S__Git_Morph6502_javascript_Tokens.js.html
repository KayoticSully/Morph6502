 35 //-----------------------------------
 36 // All Tokens / Terminals RegEx
 37 //-----------------------------------
 38 // RegEx's use the start symbol "^" since
 39 // each regex will be run to find the token
 40 // at the start of the source string.  No
 41 // end symbol "$" is used since there may
 42 // be source code after the symbol.
 43 const
 44     R_INT             = /^int/,
 45     R_CHAR            = /^char/,
 46     R_DIGIT           = /^[0-9]/,
 47     R_CHARACTER       = /^[a-z]/,
 48     R_BRACE_OPEN      = /^[{]/,
 49     R_BRACE_CLOSE     = /^[}]/,
 50     R_EPSILON         = /^[]/, // ???
 51     R_PAREN_OPEN      = /^[(]/,
 52     R_PAREN_CLOSE     = /^[)]/,
 53     R_EQUALS          = /^[=]/,
 54     R_P               = /^[P]/,
 55     R_QUOTE           = /^["|']/,
 56     R_PLUS            = /^[+]/,
 57     R_MINUS           = /^[-]/,
 58     R_$               = /^[$]/,
 59     R_NEW_LINE        = /^(\n)(\r)?/;
 60     R_SPACE           = /^\s/;
 61 
 62 //-----------------------------------
 63 // The Magic Lexer Object
 64 //-----------------------------------
 65 // Order is (somewhat) important
 66 var Tokens = {
 67     // Reserved words need to go first, otherwise each
 68     // character will get identified as T_CHARACTER tokens
 69     T_INT           : { pattern: R_INT,         length: 3 },
 70     T_CHAR          : { pattern: R_CHAR,        length: 4 },
 71     // Now all other symbols
 72     T_DIGIT         : { pattern: R_DIGIT,       length: 1 },
 73     T_CHARACTER     : { pattern: R_CHARACTER,   length: 1 },
 74     T_BRACE_OPEN    : { pattern: R_BRACE_OPEN,  length: 1 },
 75     T_BRACE_CLOSE   : { pattern: R_BRACE_CLOSE, length: 1 },
 76     // Not sure if we need epsilon here...
 77  // T_EPSILON       : { pattern: R_EPSILON,     length 0 },
 78     T_PAREN_OPEN    : { pattern: R_PAREN_OPEN,  length: 1 },
 79     T_PAREN_CLOSE   : { pattern: R_PAREN_CLOSE, length: 1 },
 80     T_EQUALS        : { pattern: R_EQUALS,      length: 1 },
 81     T_P             : { pattern: R_P,           length: 1 },
 82     T_QUOTE         : { pattern: R_QUOTE,       length: 1 },
 83     T_PLUS          : { pattern: R_PLUS,        length: 1 },
 84     T_MINUS         : { pattern: R_MINUS,       length: 1 },
 85     CT_NEW_LINE     : { pattern: R_NEW_LINE,    length: 1 },
 86     CT_SPACE        : { pattern: R_SPACE,       length: 1 },
 87     T_$             : { pattern: R_$,           length: 1 }
 88 }
