 12 //----------------------------
 13 // Lexer is a full closure to
 14 // encapulate the Lexer's
 15 // inner workings
 16 //----------------------------
 17 var Lexer = function() {
 18     
 19     var currentLine = 1;
 20     var errors = new Array();
 21     
 22     /**
 23     * @name Lexer
 24     * @param {String} source The source code to convert into tokens
 25     * @returns {Array} An array of tokens
 26     * 
 27     * @description Traverses the source code input to verify and build a token stream.
 28     */
 29     function Lexer(src) {
 30         // We can assume that the src input is already cleaned
 31         
 32         // make sure values are reset
 33         currentLine = 1;
 34         errors = new Array();
 35         
 36         // start processing source
 37         var results = {
 38             tokens : process(src),
 39             errors : errors
 40         }
 41         
 42         // check for errors
 43         return results;
 44     }
 45     
 46     /**
 47      * @name process
 48      * @param {String} src Source code to process.
 49      * @returns {Array} Stream of tokens found in source
 50      *
 51      * @description Each recursive itteration will have one less token removed from the front of the source.
 52      */
 53     function process(src) {
 54         // get next token info
 55         var tokenType   = checkToken(src);
 56         var length      = getTokenLength(tokenType);
 57         
 58         // string minus found token or error
 59         var progress    = src.substr(length); 
 60         
 61         // create new token up here
 62         // so it can be used throughout
 63         // the function
 64         var token       = new Token();
 65         
 66         // increment line number at
 67         // the end of each line
 68         switch(tokenType) {
 69             case CT_NEW_LINE:
 70                 currentLine++;
 71                 // return next token to remove this from the token stream
 72                 return process(progress);
 73             break;
 74             
 75             // error
 76             case null:
 77                 // add error count to the current line
 78                 if(errors[currentLine] == undefined) {
 79                     errors[currentLine] = 1;
 80                 } else {
 81                     errors[currentLine]++;
 82                 }
 83             break;
 84             
 85             default:
 86                 token.type = tokenType;
 87                 token.line = currentLine;
 88                 token.value = src.substr(0, length);
 89             break;
 90         }
 91         
 92         if(src.length > 1) {
 93             // concat is used here so arrays do not become nested
 94             return new Array(token).concat(process(progress)); // YAY RECURSION!
 95         } else {
 96             return token;
 97         }
 98     }
 99     
100     /**
101      * @name checkToken
102      * @param {String} str A chunk of source code to be tested for a token at the start.
103      * @returns {String} A token constant that represents the type of token found.
104      *
105      * @description Finds a token at the beginning of the source input.
106      */
107     function checkToken(str) {
108         // for each possible token
109         for(token in Tokens) {
110             // test to see if the next token is this one
111             var pattern = Tokens[token].pattern;
112             if(pattern.test(str)) {
113                 return token;
114             }
115         }
116         
117         // no supported token was found
118         return null;
119     }
120     
121     /**
122      * @name getTokenLength
123      * @param {String} tokenType Type of token to find the length of
124      * @return {Integer} length of token or 1 if token is not found
125      *
126      * @description Gets the length of a token.
127      */
128     function getTokenLength(tokenType) {
129         if(tokenType in Tokens) {
130             return Tokens[tokenType].length;
131         } else {
132             return 1;
133         }
134     }
135     
136     //----------------------------
137     // Return the Lexer so we
138     // have full closure
139     //----------------------------
140     return Lexer;
141 }
