 15 function trim(str) // Use a regular expression to remove leading and trailing spaces.
 16 {
 17     // I removed the whitespace on either side of the OR "|" operator.
 18     // It was messing up if the only space was a newline character.
 19     return str.replace(/^\s+|\s+$/g, "");
 20     /* 
 21 	Huh?  Take a breath.  Here we go:
 22 	- The "|" separates this into two expressions, as in A or B.
 23 	- "^\s+" matches a sequence of one or more whitespace characters at the beginning of a string.
 24         - "\s+$" is the same thing, but at the end of the string.
 25         - "g" makes is global, so we get all the whitespace.
 26         - "" is nothing, which is what we replace the whitespace with.
 27     */
 28 }
