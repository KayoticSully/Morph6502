 12 /**
 13  * @name displayOutput
 14  */
 15 function displayOutput() {
 16     $('#productions').addClass('hidden');
 17     $('#output, #clearOutput').removeClass('hidden');
 18 }
 19 
 20 /**
 21  * @name clearOutput
 22  * 
 23  * @description Completely clears output and shows productions pane
 24  */
 25 function clearOutput() {
 26     $('#productions').removeClass('hidden');
 27     $('#output, #clearOutput').addClass('hidden');
 28     $('.lineselect').removeClass('lineselect');
 29 }
 30 
 31 /**
 32  * @name resetOutput
 33  * 
 34  * @description Only clears output pane and line highlighting
 35  */
 36 function resetOutput() {
 37     // reset line highlighting and output pane
 38     $('.lineselect').removeClass('lineselect');
 39     $('#output').html('');
 40 }
 41 
 42 /**
 43  * @name log
 44  * @param {String} line A line to add to output list
 45  * @param {String} type Classes to add to output list item
 46  *
 47  * @description Controls compiler output
 48  */
 49 function log(line, type) {
 50     
 51     if(type === undefined) type = '';
 52     
 53     var str =   '<li class="' + type + '">' +
 54                     line +
 55                 '</li>';
 56     $('#output').append(str);
 57 }

 59 /**
 60  * @name displayTokenErrors
 61  * @param {Object} tokenErrors An object with line numbers for keys and the number of errors on that line as values
 62  *
 63  * @description Handles the display of errors found during Lexing
 64  */
 65 function displayTokenErrors(tokenErrors) {
 66     for(lineNumber in tokenErrors) {
 67         var errors = tokenErrors[lineNumber];
 68         var amountModifier = 's';
 69         var preposition = 'are';
 70         
 71         if(errors == 1) {
 72             amountModifier = ' ';
 73             preposition = 'is ';
 74         }
 75         
 76         var str =   'There ' + preposition + ' ' +
 77                     errors + ' unknown character' + amountModifier +
 78                     ' on line ' + lineNumber;
 79         
 80         log(str, 'error');
 81         
 82         var index = parseInt(lineNumber) - 1
 83         $('.codelines .lineno:eq(' + index + ')').addClass('lineselect');
 84     }
 85     
 86     displayOutput();
 87 }
